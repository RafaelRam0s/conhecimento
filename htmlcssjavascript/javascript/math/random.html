Random

Com o random podemos sortear um número maior que 0 e menor que 1

function numAleatorio() { window.alert(Math.random()) }

Com um pouco mais de código é possível ajustar para que sejam sorteados números inteiros

function sortearNumeroInteiro(min, max) { let numero_max = max - min + 1 let numero_max_digitos = ('' + Math.abs(numero_max)).length let numero_sorteado = parseInt(Math.random() * (10 ** numero_max_digitos)) + 1 let numero_maximo_sorteado = '' for (let x = 0; numero_max_digitos - x != 0; x++) { numero_maximo_sorteado += 9 } numero_maximo_sorteado = Number(numero_maximo_sorteado) + 1 numero_sorteado = (numero_max * numero_sorteado) / numero_maximo_sorteado if (numero_sorteado % 1 > 0) { numero_sorteado = parseInt(numero_sorteado) + 1 } else if (numero_sorteado % 1 < 0) { numero_sorteado = parseInt(numero_sorteado) - 1 } numero_sorteado = numero_sorteado + min - 1 window.alert(numero_sorteado) } function sortearNumerosEntre(min, max) { let numMin = 0, numMax = 0; if (min < max) { numMin = min; numMax = max; } else { numMin = max; numMax = min; } let raio = Math.abs(numMax - numMin + 1); let numSorteado = numMin + Math.floor(Math.random() * raio); window.alert(numSorteado) }
