Trunc

Não é arredondado, simplesmente casas decimais são retirados do número

function usoDoTrunc() { window.alert( 'Como explicado as casas decimais somem: \n' + 'Número 0.99: ' + Math.trunc(0.99) + '\n' + 'Número 0.12: ' + Math.trunc(0.12) + '\n' + 'Número 123: ' + Math.trunc(123) + '\n' + 'Número 99.5: ' + Math.trunc(99.5) + '\n' + 'Número 4.9: ' + Math.trunc(4.9) + '\n' + 'Número 4.5: ' + Math.trunc(4.5) + '\n' + 'Número 4.49: ' + Math.trunc(4.49) + '\n' + 'Número 4.1: ' + Math.trunc(4.1) + '\n' + 'Número 1.000000000000000000000000000000000001: ' + Math.trunc(1.000000000000000000000000000000000001) + '\n' + 'Número 1.999999999999999999999999999999999999: ' + Math.trunc(1.999999999999999999999999999999999999) ) }

Ceil

É feito um arredondamento somando +1 a parte inteira, se existir alguma casa decimal

function usoDoCeil() { window.alert( 'Como explicado as casas decimais somem: \n' + 'Número 0.99: ' + Math.ceil(0.99) + '\n' + 'Número 0.12: ' + Math.ceil(0.12) + '\n' + 'Número 123: ' + Math.ceil(123) + '\n' + 'Número 99.5: ' + Math.ceil(99.5) + '\n' + 'Número 4.9: ' + Math.ceil(4.9) + '\n' + 'Número 4.5: ' + Math.ceil(4.5) + '\n' + 'Número 4.49: ' + Math.ceil(4.49) + '\n' + 'Número 4.1: ' + Math.ceil(4.1) + '\n' + 'Número 1.000000000000000000000000000000000001: ' + Math.ceil(1.000000000000000000000000000000000001) + '\n' + 'Número 1.999999999999999999999999999999999999: ' + Math.ceil(1.999999999999999999999999999999999999) ) }

Floor

É feito um arredondamento deixando só a parte inteira, se existir alguma casa decimal ela é retirada, parece muito com o trunc@@@@@

function usoDoFloor() { window.alert( 'Como explicado as casas decimais somem: \n' + 'Número 0.99: ' + Math.floor(0.99) + '\n' + 'Número 0.12: ' + Math.floor(0.12) + '\n' + 'Número 123: ' + Math.floor(123) + '\n' + 'Número 99.5: ' + Math.floor(99.5) + '\n' + 'Número 4.9: ' + Math.floor(4.9) + '\n' + 'Número 4.5: ' + Math.floor(4.5) + '\n' + 'Número 4.49: ' + Math.floor(4.49) + '\n' + 'Número 4.1: ' + Math.floor(4.1) + '\n' + 'Número 1.000000000000000000000000000000000001: ' + Math.floor(1.000000000000000000000000000000000001) + '\n' + 'Número 1.999999999999999999999999999999999999: ' + Math.floor(1.999999999999999999999999999999999999) ) }

Round

É feito um arredondamento, se a casa decimal for menor que 5 age como floor (arredonda deixando só a parte inteira), se maior que 5 age como ceil (arredonda somando +1 na parte inteira)

function usoDoRound() { window.alert( 'Como explicado as casas decimais somem: \n' + 'Número 0.99: ' + Math.round(0.99) + '\n' + 'Número 0.12: ' + Math.round(0.12) + '\n' + 'Número 123: ' + Math.round(123) + '\n' + 'Número 99.5: ' + Math.round(99.5) + '\n' + 'Número 4.9: ' + Math.round(4.9) + '\n' + 'Número 4.5: ' + Math.round(4.5) + '\n' + 'Número 4.49: ' + Math.round(4.49) + '\n' + 'Número 4.1: ' + Math.round(4.1) + '\n' + 'Número 1.000000000000000000000000000000000001: ' + Math.round(1.000000000000000000000000000000000001) + '\n' + 'Número 1.999999999999999999999999999999999999: ' + Math.round(1.999999999999999999999999999999999999) ) }
